# A+B

```c++
#include<iostream>
#include<vector>
using namespace std;
//c=a+b
vector<int> add(vector<int>&A,vector<int>&B){
    vector<int> c;
    int t=0;
    for(int i=0;i<A.size()||i<B.size();i++){
        if(i<A.size()) t+=A[i];
        if(i<B.size()) t+=B[i];
        c.push_back(t%10);
        t/=10;
    }
    if(t) c.push_back(1);
    return c;
}
int main(){
    string a,b;//大整数用string输入
    cin>>a>>b;//a="123456"
    vector<int> A,B;//转换成数组储存
    for(int i=a.size()-1;i>=0;i--) A.push_back(a[i]-'0');
    for(int i=b.size()-1;i>=0;i--) B.push_back(b[i]-'0');
    auto c=add(A,B);
    for(int i=c.size()-1;i>=0;i--) cout<<c[i];
    return 0;
}
```

# A/a

唯一不同是除法是从高位开始的运算，因此要反转一次，然后再进行消0处理，最后正常输出。

```c++
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<int> div(vector<int>&A,int b,int& r){
    vector<int> c;
    for(int i=A.size()-1;i>=0;i--){
        r=r*10+A[i];
        c.push_back(r/b);
        r%=b;
    }
    reverse(c.begin(),c.end());
    while(c.size()>1&&c.back()==0) c.pop_back();
    return c;
}
int main(){
    string a;
    int b,r=0;
    cin>>a;
    cin>>b;
    vector<int> A;
    for(int i=a.size()-1;i>=0;i--) A.push_back(a[i]-'0');
    auto c=div(A,b,r);
    for(int i=c.size()-1;i>=0;i--) cout<<c[i];
    cout<<endl<<r;
    return 0;
}
```

# 前缀和

**把前缀和与hashmap刻入DNA**

**求个数就存个数，哨兵存（0，1） 求长度就存下标，哨兵存（0，-1）**

原数组a1，a2 ....an

前缀和就是求前i个元素的和 s[i]=a1+a2+..+ai

怎么求：dp **s[i]=s[i-1]+ai**

有什么用：快速的求出一段区间的和

求[l,r]的和，就等于s[r]-s[l-1] ,**特殊的，定义s[0]=0;且数组要从a1开始而不是a0**

如果是从0开始，就是**s[r+1]-s[l]**，dp **s[i+1]=s[i]+nums[i]**

## 二维前缀和

```c++
//当从0开始时
```

# 差分

对于原数组an，构造bn，使得an数组是bn数组的前缀和，此时

**ai=b1+b2+...+bi**

一维：b1=a1 b2=a2-a1 b3=a3-a2....bn=an-an-1

```c++
#include<iostream>
using namespace std;
int n,m;
const int N=100010;
int a[N],b[N];
void insert(int l,int r,int c){
    b[l]+=c;
    b[r+1]-=c;
}
int main(){
    cin>>n>>m;
    for(int i=1;i<=n;i++) cin>>a[i];
    //初始化b[i]，如果是在0数组上进行插入，就不需要初始化
    for(int i=1;i<=n;i++) insert(i,i,a[i]);
  //这样插入构造出来的b[i]就是a的差分数组。b[1]=a[1],b[2]=a[2]-a[1]....
    while(m--){
        int l,r,c;
        cin>>l>>r>>c;
        insert(l,r,c);
    }
    //求前缀和
    for(int i=1;i<=n;i++) b[i]+=b[i-1];
    for(int i=1;i<=n;i++) cout<<b[i]<<" ";
    return 0;
}
```

## 二维差分

给子矩阵加上一个值，**数组从1开始**

```c++
#include<iostream>
using namespace std;
const int N=1010;
int a[N][N],b[N][N];
int n,m,q;
void insert(int x1,int y1,int x2,int y2,int c){
    b[x1][y1]+=c;
    b[x2+1][y1]-=c;
    b[x1][y2+1]-=c;
    b[x2+1][y2+1]+=c;
}
int main(){
    cin>>n>>m>>q;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            cin>>a[i][j];
        }
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            insert(i,j,i,j,a[i][j]);
        }
    }
    while(q--){
        int x1,y1,x2,y2,c;
        cin>>x1>>y1>>x2>>y2>>c;
        insert(x1,y1,x2,y2,c);
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            b[i][j]+=b[i-1][j]+b[i][j-1]-b[i-1][j-1];
            cout<<b[i][j]<<" ";
        }
        cout<<endl;
    }
}
```
